# Share Mix via URL Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let users share a generated mix via a URL that recipients can open to see pre-filled artists/tags and optionally auto-generate the same mix.

**Architecture:** Share params encoded as URL query params. `sessionStorage` bridges share params through OAuth redirect. Three new functions in `ui.js`: `buildShareURL()`, `handleShare()`, `loadFromShareParams()`. Two new globals track last mix type and tags for reliable URL construction.

**Tech Stack:** Vanilla JS, `URLSearchParams`, `navigator.share` / `navigator.clipboard`, `sessionStorage`

**Spec:** `docs/superpowers/specs/2026-03-21-share-mix-design.md`

```js
function buildShareURL() {
  const params = new URLSearchParams();
  if (lastMixType === 'tag' && lastMixTags.length) {
    params.set('tags', lastMixTags.join(','));
    if (tracksPerTag !== 5) params.set('tcount', tracksPerTag);
  } else {
    const active = artists.filter(Boolean);
    if (!active.length) return null;
    params.set('artists', active.map(a => a.name).join(','));
    if (trackMode !== 'top') params.set('mode', trackMode);
    if (tracksPerArtist !== 5) params.set('count', tracksPerArtist);
  }
  params.set('auto', '1');
  return REDIRECT_URI + '?' + params.toString();
}
```

- [ ] **Step 2: Add `handleShare()` function**

```js
async function handleShare() {
  const url = buildShareURL();
  if (!url) return;
  try {
    if (navigator.share) {
      await navigator.share({ title: 'Check out this SpotiMix', url: url });
      return;
    }
  } catch (e) {
    // User cancelled native share or it failed — fall through to clipboard
    if (e.name === 'AbortError') return;
  }
  try {
    await navigator.clipboard.writeText(url);
    showToast('Link copied!');
  } catch {
    showToast('Could not copy link');
  }
}
```

```js
async function loadFromShareParams() {
  // Check sessionStorage first (persisted through OAuth redirect), then URL
  const raw = sessionStorage.getItem('share_params') || window.location.search;
  sessionStorage.removeItem('share_params');
  if (!raw || raw === '?') return;

  const p = new URLSearchParams(raw);
  const artistsParam = p.get('artists');
  const tagsParam = p.get('tags');
  if (!artistsParam && !tagsParam) return;

  // Clean URL
  window.history.replaceState({}, '', REDIRECT_URI);

  if (artistsParam) {
    // ── Artist Mix path ──
    const names = artistsParam.split(',').slice(0, 3).map(s => s.trim()).filter(Boolean);
    let found = 0;
    for (let i = 0; i < names.length; i++) {
      try {
        const data = await spGet('/search?q=' + encodeURIComponent(names[i]) + '&type=artist&limit=1');
        const a = data.artists && data.artists.items && data.artists.items[0];
        if (a) {
          artists[i] = {
            name: a.name,
            image: (a.images && a.images[0] && a.images[0].url) || '',
            sub: a.followers ? a.followers.total.toLocaleString() + ' followers' : '',
          };
          found++;
        } else {
          showToast("Couldn't find " + names[i]);
        }
      } catch {
        showToast("Couldn't find " + names[i]);
      }
    }
    if (!found) { showToast("Couldn't load shared mix"); return; }
    renderAllSlots();
    updateSuggest();

    // Apply mode and count from params
    const mode = p.get('mode');
    if (mode && ['top', 'deep', 'mix', 'discovery'].includes(mode)) {
      setModeByName(mode);
    }
    const count = parseInt(p.get('count'), 10);
    if (count >= 1 && count <= 10) {
      tracksPerArtist = count;
      document.getElementById('track-count').textContent = tracksPerArtist;
    }

    if (p.get('auto') === '1') generate();
  } else {
    // ── Tag Mix path ──
    const tags = tagsParam.split(',').slice(0, 3).map(s => s.trim()).filter(Boolean);
    if (!tags.length) return;

    tags.forEach(t => selectedGenres.add(t));

    const tcount = parseInt(p.get('tcount'), 10);
    if (tcount >= 1 && tcount <= 10) {
      tracksPerTag = tcount;
    }

    setEntry('browse');

    if (p.get('auto') === '1') generateTagMix();
  }
}
```
